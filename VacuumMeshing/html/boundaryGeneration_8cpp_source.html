131 void changeMeshBasis(Eigen::MatrixXd &verts, Eigen::Vector3d newOrigin,
132  Eigen::Matrix3d newBasis, Eigen::Vector3d oldOrigin,
133  Eigen::Matrix3d oldBasis) {
134  // Inverse the new basis matrix
135  Eigen::Matrix3d newBasisInverse = newBasis.inverse();
136  // We assume we are changing basis from bog standard cartesian coords.
137  // i.e. an origin of 0,0,0 and basis vectors of (1,0,0), (0,1,0), (0,0,1)
138  // Eigen::Matrix3d oldBasis(Eigen::Matrix3d::Identity());
139  // Eigen::Vector3d oldOrigin(Eigen::Vector3d::Zero());
140  // Do row wise operations on
141  for (int i = 0; i < verts.rows(); i++) {
142  verts.row(i) = newBasisInverse *
143  (oldOrigin - newOrigin + (oldBasis * verts.row(i).transpose()));
144  }
145 }

153 bool getBasisMatrix(Eigen::Matrix3d &basisMatrix,
154  Eigen::Matrix3d &planePoints) {
155  // Basis vectors X Y and Z
156  Eigen::Vector3d X, Y, Z;
157 
158  // Vectors between the 3 points that will define our plane
159  Eigen::Vector3d AB = (planePoints.row(1) - planePoints.row(0));
160  Eigen::Vector3d AC = (planePoints.row(2) - planePoints.row(0));
161 
162  // Normalise basis vectors so that their magnitude is one, relative to
163  // original basis
164  X = AB.normalized();
165  Z = (AC.cross(AB)).normalized();
166  Y = (X.cross(Z)).normalized();
167  // Add our basis vectors to 'basisMatrix' matrix
168  basisMatrix.col(0) << X;
169  basisMatrix.col(1) << Y;
170  basisMatrix.col(2) << Z;
171 
172  return true;
173 }

296 void genSquare(Eigen::MatrixXd &verts, Eigen::MatrixXi &elems, double length,
297  int subdivisions) {
298  // Add second order capability!
299 
300  //
301  double increment = (double)(length / subdivisions);
302  // Offset so that the square mesh is symmetric about the x and y axes
303  double offset = length / 2;
304 
305  long int row_num = 0;
306  for (int i = 0; i <= subdivisions; i++) {
307  verts.row(row_num) << -offset + (i * increment), -offset;
308  row_num++;
309  }
310 
311  for (int i = 1; i <= subdivisions; i++) {
312  verts.row(row_num) << offset, (i * increment) - offset;
313  row_num++;
314  }
315 
316  for (int i = 1; i <= subdivisions; i++) {
317  verts.row(row_num) << offset - (i * increment), offset;
318  row_num++;
319  }
320 
321  for (int i = 1; i <= subdivisions; i++) {
322  verts.row(row_num) << -offset, offset - (i * increment);
323  row_num++;
324  }
325 
326  for (int i = 0; i < (4 * subdivisions) - 1; i++) {
327  elems(i, 0) = i;
328  elems(i, 1) = i + 1;
329  }
330  elems(((4 * subdivisions) - 1), 0) = 0;
331  elems(((4 * subdivisions) - 1), 1) = (4 * subdivisions) - 1;
332 }
