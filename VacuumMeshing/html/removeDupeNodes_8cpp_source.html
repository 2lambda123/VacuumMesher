220  Eigen::MatrixXd &V2, Eigen::MatrixXi &F2) {
221  // Instantiate an rTree. Using a rTree data structure significantly reduces
222  // the amount of time taken to discover duplicate nodes.
223  nodeTree rtree;
224 
225  // This is a map which helps us keep track of the node id's of the duplicate
226  // nodes. For example, if Node 4 in the surface mesh and Node 6 in the vacuum
227  // mesh are the same, "id_map[4]" will return 6
228  std::map<unsigned int, unsigned int> id_map;
229 
230  createTree(rtree, V1, tol);
231 
232  int starting_elems = F1.rows();
233  int node_indent = V1.rows();
234 
235  for (int node_id = 0; node_id < V2.rows(); node_id++) {
236  Box node_box(V2.row(node_id), tol, node_indent);
237  // If there are no matches in the vacuumMesh, then we add the point
238  if (!searchTree(rtree, tol, id_map, node_indent, node_box)) {
239  id_map[node_id] = hits[0];
240  continue;
241  }
242 
243  id_map[node_id] = node_indent;
244  node_indent++;
245  }
246 
247  V1.conservativeResize(node_indent, 3);
248 
249  nodeTree::Iterator it;
250  for (rtree.GetFirst(it); !rtree.IsNull(it); rtree.GetNext(it)) {
251  double boundsMin[3] = {0., 0., 0.};
252  double boundsMax[3] = {0., 0., 0.};
253  it.GetBounds(boundsMin, boundsMax);
254 
255  double centre[3];
256  for (int i = 0; i < 3; i++) {
257  centre[0] = boundsMin[0] + ((boundsMax[0] - boundsMin[0]) / 2);
258  centre[1] = boundsMin[1] + ((boundsMax[1] - boundsMin[1]) / 2);
259  centre[2] = boundsMin[2] + ((boundsMax[2] - boundsMin[2]) / 2);
260  }
261  V1.row (*it)(0) = centre[0];
262  V1.row (*it)(1) = centre[1];
263  V1.row (*it)(2) = centre[2];
264  }

267

269  F1.row(elem_id + starting_elems)(0) = id_map[F2.row(elem_id)(0)];
270  F1.row(elem_id + starting_elems)(1) = id_map[F2.row(elem_id)(1)];
271  F1.row(elem_id + starting_elems)(2) = id_map[F2.row(elem_id)(2)];
272  }
